This schema describes the Boston Area Activity Groups database, supporting residents, activity groups, events, and sessions.


Entity: resident
- _resident_id_ INTEGER PRIMARY KEY
- name TEXT NOT NULL
- email TEXT UNIQUE NOT NULL -- format: valid email
- phone_number TEXT UNIQUE -- format: optional, digits with optional dashes/spaces
- interests TEXT -- comma-separated list or JSON string of enums
- date_of_birth TEXT -- format: YYYY-MM-DD
- profile_image TEXT -- file path or image URL

Entity: activity_group
- _name_ TEXT PRIMARY KEY
- category TEXT -- e.g., sports, networking, arts
- description TEXT
- founding_date TEXT -- format: YYYY-MM-DD
- website TEXT -- URL
- email TEXT -- contact email
- phone_number TEXT -- contact number
- social_media_links TEXT -- stored as a JSON string, e.g. {"facebook": "...", "x": "..."}
- is_active BOOLEAN DEFAULT 1 -- 1 = active, 0 = inactive
- total_members INTEGER DEFAULT 0 -- auto-updated via application logic
- event_frequency TEXT -- enum: weekly, monthly, etc.
- membership_fee INTEGER -- amount in USD
- open_to_public BOOLEAN -- 1 = yes, 0 = no
- min_age INTEGER -- minimum age required to join

Entity: review
- _review_id_ INTEGER PRIMARY KEY
- resident_id INTEGER NOT NULL REFERENCES resident(resident_id) -- FK to resident
- activity_group_name TEXT NOT NULL REFERENCES activity_group(name) -- FK to activity_group
- content TEXT
- star_rating INTEGER CHECK (star_rating BETWEEN 1 AND 5)
- review_date TEXT -- format: YYYY-MM-DD
- is_verified BOOLEAN DEFAULT 0 -- 1 = verified review

Entity: event
- _event_id_ INTEGER PRIMARY KEY
- activity_group_name TEXT NOT NULL REFERENCES activity_group(name) -- FK to activity_group
- date TEXT NOT NULL -- format: YYYY-MM-DD
- location_id INTEGER REFERENCES location(location_id) -- FK to location
- max_participants INTEGER
- cost INTEGER -- amount in USD
- registration_required BOOLEAN -- 1 = required, 0 = open
- registration_deadline TEXT -- format: YYYY-MM-DD

Entity: session
- _session_id_ INTEGER PRIMARY KEY
- activity_group_name TEXT NOT NULL REFERENCES activity_group(name) -- FK to activity_group
- event_id INTEGER REFERENCES event(event_id) -- FK to event (if this session is tied to an event)
- date TEXT -- format: YYYY-MM-DD
- attendance INTEGER -- number of attendees
- agenda TEXT

Entity: location
- _location_id_ INTEGER PRIMARY KEY
- address TEXT
- city TEXT
- state TEXT
- zip_code TEXT -- 5-digit ZIP

-- ASSOCIATIVE ENTITIES --

Entity: member (associative)
- resident_id INTEGER REFERENCES resident(resident_id) -- FK
- activity_group_name TEXT REFERENCES activity_group(name) -- FK
- join_date TEXT -- format: YYYY-MM-DD
- role TEXT -- e.g., member, organizer, treasurer
- PRIMARY KEY (resident_id, activity_group_name)

Entity: hosts (associative)
- activity_group_name TEXT REFERENCES activity_group(name) -- FK
- session_id INTEGER REFERENCES session(session_id) -- FK
- PRIMARY KEY (activity_group_name, session_id)

Entity: prerequisite (associative, self-referencing)
- event_id INTEGER REFERENCES event(event_id) -- FK
- prerequisite_event_id INTEGER REFERENCES event(event_id) -- FK
- minimum_performance INTEGER -- performance rating or score
- qualification_period INTEGER -- days the qualification remains valid
- is_waiver_allowed BOOLEAN -- 1 = yes, 0 = no
- PRIMARY KEY (event_id, prerequisite_event_id)

-- CONSTRAINTS & FORMATS --

UNIQUE CONSTRAINTS
-- resident.email and resident.phone_number must be unique
-- activity_group.name is unique and serves as PK

CHECK CONSTRAINTS
-- review.star_rating must be between 1 and 5
-- prerequisite.event_id must not equal prerequisite_event_id
--   CHECK (event_id != prerequisite_event_id)

NOT NULL CONSTRAINTS
-- resident.name, resident.email cannot be NULL
-- review.resident_id, review.activity_group_name cannot be NULL
-- event.activity_group_name, event.date cannot be NULL
-- member.resident_id, member.activity_group_name cannot be NULL

BOOLEAN FIELDS (stored as INTEGER 0 or 1)
-- is_active, is_verified, open_to_public, registration_required, is_waiver_allowed

DATE FORMAT (Handled in application logic)
   - All dates (e.g., date_of_birth, review_date, event.date, registration_deadline) should follow 'YYYY-MM-DD' format.
